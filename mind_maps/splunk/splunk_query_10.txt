            pattern that matches wildcard
        <<MATCHSEG1>> <<MATCHSEG2>> <<MATCHSEG3>>

        foreach test* [ eval total=total + '<<FIELD>>' ]
        foreach * [ eval <<FIELD>>='<<FIELD>>'/1024/1024/1024 ]
        foreach foo* [ eval new_<<MATCHSTR>> = <<FIELD>> + bar<<MATCHSTR>> ]
            new_X = fooX + barX
