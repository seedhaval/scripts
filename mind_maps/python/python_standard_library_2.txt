python standard library
    built in functions
        delattr( object, name )
            delete named attribute
        dict
            create dictionary
        dir( [object] )
            without argument return list of names in current local scope
            with argument, list attributes of object
        divmod( a,b )
            return [quotient, remainder] of a/b
        enumerate( iter, start=0 )
        filter( function, iterable ):
            [x for x in iterable if function(x)]
        float
            supported - +infinity, -infinity, +inf, -inf, nan
        frozenset( [iterable] )
        getattr( object, name [, def] ):
            value of name attribute of object
            return default if not present
        globals
            dictionary implementing current module namespace
